ShadowWire Browser Example

This example requires bundling with webpack, vite, or similar. Copy the WASM file to your public folder before running.

Setup

npm install @radr/shadowwire

cp node_modules/@radr/shadowwire/dist/wasm/settler_wasm_bg.wasm public/wasm/

1. Initialize WASM

import { initWASM, isWASMSupported } from '@radr/shadowwire';

if (!isWASMSupported()) {
  throw new Error('WebAssembly not supported');
}

await initWASM('/wasm/settler_wasm_bg.wasm');

2. Generate Range Proof

import { generateRangeProof } from '@radr/shadowwire';

const amount = 100000000;
const proof = await generateRangeProof(amount, 64);

console.log('Proof:', proof.proofBytes);
console.log('Commitment:', proof.commitmentBytes);
console.log('Blinding Factor:', proof.blindingFactorBytes);

3. Verify Proof

import { verifyRangeProof } from '@radr/shadowwire';

const isValid = await verifyRangeProof(
  proof.proofBytes,
  proof.commitmentBytes,
  64
);

console.log('Valid:', isValid);

4. Make a Transfer

import { ShadowWireClient } from '@radr/shadowwire';

const client = new ShadowWireClient();

const result = await client.transfer({
  sender: 'YOUR_WALLET_ADDRESS',
  recipient: 'RECIPIENT_WALLET_ADDRESS',
  amount: 0.1,
  token: 'SOL',
  type: 'internal',
  wallet: { signMessage }
});

console.log('Transaction:', result.tx_signature);
console.log('Amount Hidden:', result.amount_hidden);

5. Transfer with Custom Proof

import { ShadowWireClient, generateRangeProof, initWASM } from '@radr/shadowwire';

await initWASM();

const client = new ShadowWireClient();
const amountLamports = 100000000;

const proof = await generateRangeProof(amountLamports, 64);

const result = await client.transferWithClientProofs({
  sender: 'YOUR_WALLET_ADDRESS',
  recipient: 'RECIPIENT_WALLET_ADDRESS',
  amount: 0.1,
  token: 'SOL',
  type: 'internal',
  customProof: proof,
  wallet: { signMessage }
});

console.log('Transaction:', result.tx_signature);

6. Check Balance and Fee Info

import { ShadowWireClient } from '@radr/shadowwire';

const client = new ShadowWireClient();

const balance = await client.getBalance('YOUR_WALLET_ADDRESS', 'SOL');
console.log('Available:', balance.available / 1e9, 'SOL');

const feePercent = client.getFeePercentage('SOL');
const minimum = client.getMinimumAmount('SOL');
const breakdown = client.calculateFee(1.0, 'SOL');

console.log('Fee:', feePercent * 100, '%');
console.log('Minimum:', minimum, 'SOL');
console.log('Net amount for 1 SOL:', breakdown.netAmount);

Full React Example

import { useState, useEffect } from 'react';
import { ShadowWireClient, initWASM, isWASMSupported } from '@radr/shadowwire';

function Transfer({ userWallet, signMessage }) {
  const [client] = useState(() => new ShadowWireClient());
  const [ready, setReady] = useState(false);

  useEffect(() => {
    async function init() {
      if (isWASMSupported()) {
        await initWASM('/wasm/settler_wasm_bg.wasm');
        setReady(true);
      }
    }
    init();
  }, []);

  const handleTransfer = async () => {
    const result = await client.transfer({
      sender: userWallet,
      recipient: 'RECIPIENT_ADDRESS',
      amount: 0.1,
      token: 'SOL',
      type: 'internal',
      wallet: { signMessage }
    });
    console.log('Done:', result.tx_signature);
  };

  return (
    <button onClick={handleTransfer} disabled={!ready}>
      Send 0.1 SOL
    </button>
  );
}
